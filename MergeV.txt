export function formatCurrency(amount: number, currency: string = "USD", locale: string = "en-US"): string {

  return num.toLocaleString()
}

// ═════════════════════════════════════════════════════════════════════════

// ═════════════════════════════════════════════════════════════════════════

export function formatDate(date: Date | string | number, format: "short" | "long" | "relative" = "short"): string {
  try {
    const d = typeof date === "string" || typeof date === "number" ? new Date(date) : date
    if (isNaN(d.getTime())) return "Invalid date"
    
    if (format === "relative") {
      const now = new Date()
      const diff = now.getTime() - d.getTime()
      const minutes = Math.floor(diff / 60000)
      const hours = Math.floor(minutes / 60)
      const days = Math.floor(hours / 24)
      
      if (minutes < 1) return "just now"
      if (minutes < 60) return minutes + "m ago"
      if (hours < 24) return hours + "h ago"
      if (days < 7) return days + "d ago"
    }
    
    const options: Intl.DateTimeFormatOptions = format === "short"
      ? { month: "short", day: "numeric" }
      : { year: "numeric", month: "long", day: "numeric" }
    
    return d.toLocaleDateString("en-US", options)
  } catch {
    return "N/A"
  }
}

export function uniqueBy<T>(arr: T[], key: keyof T): T[] {
  const seen = new Set()
  return arr.filter(item => {
    const value = item[key]
    if (seen.has(value)) return false
    seen.add(value)
    return true
  })
}

  return arr.reduce((acc, item) => {
    const k = String(item[key])
    if (!acc[k]) acc[k] = []
    acc[k].push(item)
    return acc
  }, {} as Record<string, T[]>)
}

      fn(...args)
      inThrottle = true
      setTimeout(() => { inThrottle = false }, limit)

  }

//  ASYNC UTILITIES — RETRY & TIMEOUT
// ═════════════════════════════════════════════════════════════════════════

export async function withRetry<T>(
  fn: () => Promise<T>,
  retries: number = 3,
  delay: number = 1000,
  onRetry?: (attempt: number, error: Error) => void
): Promise<T> {
  let lastError: Error
  
  for (let attempt = 1; attempt <= retries; attempt++) {
    try {
      return await fn()
    } catch (e) {
      lastError = e instanceof Error ? e : new Error(String(e))
      onRetry?.(attempt, lastError)
      if (attempt === retries) break
      await new Promise(resolve => setTimeout(resolve, delay * attempt))
    }
  }
  throw lastError!
}

export function withTimeout<T>(promise: Promise<T>, ms: number, timeoutError: string = "Request timeout"): Promise<T> {
  return Promise.race([
    promise,
    new Promise<T>((_, reject) => setTimeout(() => reject(new Error(timeoutError)), ms))
  ])
}

    g = parseInt(hex[2] + hex[2], 16)
    b = parseInt(hex[3] + hex[3], 16)

    b = parseInt(hex.slice(5, 7), 16)
  }
  return "rgba(" + r + "," + g + "," + b + "," + alpha + ")"
}

async function fetchWithRetry(
  url: string,
  body: Record<string, unknown>,
  retries: number = 4,
  onRetry?: (attempt: number, error: Error) => void
): Promise<Response> {
  
  for (let attempt = 1; attempt <= retries; attempt++) {
    try {
      const response = await fetch(url, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(body),
        signal: AbortSignal.timeout(30000) // 30s timeout
      })
      
      if (response.ok) return response
      
      // Handle rate limiting & server errors with exponential backoff
      if (response.status === 429 || response.status >= 500) {
        const delay = Math.pow(2, attempt - 1) * 800 + Math.random() * 400
        onRetry?.(attempt, new Error("HTTP " + response.status + " — retrying in " + Math.round(delay) + "ms"))
        await new Promise(resolve => setTimeout(resolve, delay))
        continue
      }
      
      // Parse error response
      const errorData: GeminiErrorResponse = await response.json().catch(() => ({}))
      throw new Error(errorData.error?.message || "HTTP " + response.status)
      
    } catch (error) {
      const err = error instanceof Error ? error : new Error(String(error))
      
      // Don't retry on client errors
      if (err.message.includes("AbortError") || err.message.includes("400")) {
        throw err
      }
      
      onRetry?.(attempt, err)
      
      if (attempt === retries) {
        throw new Error("Max retries exceeded: " + err.message)
      }
      
      // Exponential backoff
      const delay = Math.pow(2, attempt - 1) * 800
      await new Promise(resolve => setTimeout(resolve, delay))
    }
  }
  
  throw new Error("Unexpected fetch failure")
}

export function truncateToTokenLimit(
  text: string,

  reserve: number = 2000

  return text.slice(0, maxLength) + "\n\n[...truncated for context limit]"

// ═════════════════════════════════════════════════════════════════════════
//  HISTORY MANAGEMENT — UNDO/REDO WITH LIMIT
// ═════════════════════════════════════════════════════════════════════════

interface HistoryState<T> {
  past: T[]
  future: T[]
  max: number
  push: (state: T) => void
  undo: () => T | undefined
  redo: () => T | undefined
  canUndo: boolean
  canRedo: boolean
}

function createHistory<T>(max: number = 20): HistoryState<T> {
  return {
    past: [],
    future: [],
    max,
    push(state: T) {
      this.past.push(JSON.parse(JSON.stringify(state)))
      if (this.past.length > this.max) this.past.shift()
      this.future = []
    },
    undo() {
      if (this.past.length === 0) return undefined
      const current = this.past.pop()
      if (current) {
        this.future.push(JSON.parse(JSON.stringify(current)))
        return current
      }
      return undefined
    },
    redo() {
      if (this.future.length === 0) return undefined
      const next = this.future.pop()
      if (next) {
        this.past.push(JSON.parse(JSON.stringify(next)))
        return next
      }
      return undefined
    },
    get canUndo() { return this.past.length > 0 },
    get canRedo() { return this.future.length > 0 }
  }
}

// Fix edge references after node creation (safety)
function fixEdgeReferences(state: CanvasState): CanvasState {
  if (!validateArray(state.nodes) || state.nodes.length < 2) return state
  
  const nodeIds = state.nodes.map(n => n?.id).filter(Boolean) as string[]
  const edges = (state.edges || []).map((edge, i) => {
    if (!edge) return edge
    return {
      ...edge,
      source: nodeIds[Math.min(i, nodeIds.length - 2)] || edge.source,
      target: nodeIds[Math.min(i + 1, nodeIds.length - 1)] || edge.target
    }
  })
  
  return { ...state, edges }
}

        if (!id) return
        s.canvasState.edges = (s.canvasState.edges || []).filter(e => e?.id !== id)
        s.canvasState.history.push(s.canvasState)
      }),
      
      setViewport: (viewport) => set(s => {

      }),

        s.canvasState.selectedIds = ids
      }),
      
      copySelection: () => set(s => {

        const nodes = (s.canvasState.nodes || []).filter(n => n && ids.includes(n.id))
        const edges = (s.canvasState.edges || []).filter(e => e && ids.includes(e.source) && ids.includes(e.target))
        s.canvasState.clipboard = {
          nodes: JSON.parse(JSON.stringify(nodes)),
          edges: JSON.parse(JSON.stringify(edges))
        }
      }),
      
      pasteSelection: (offset = { x: 30, y: 30 }) => set(s => {
        if (!s.canvasState.clipboard) return
        const idMap: Record<string, string> = {}
        
        const newNodes = (s.canvasState.clipboard.nodes || []).map(n => {
          if (!n) return null
          const newId = uid("node")
          idMap[n.id] = newId

            ...n,
            id: newId,
            x: n.x + offset.x,
            y: n.y + offset.y,
            zIndex: ((s.canvasState.nodes || []).reduce((max, nd) => Math.max(max, nd?.zIndex || 0), 0) + 1),
            metadata: {
              createdAt: new Date().toISOString(),
              updatedAt: new Date().toISOString()
            }
          }
        }).filter(Boolean) as CanvasNode[]
        
        const newEdges = (s.canvasState.clipboard.edges || []).map(e => {
          if (!e || !idMap[e.source] || !idMap[e.target]) return null
          return { ...e, id: uid("edge"), source: idMap[e.source], target: idMap[e.target] }
        }).filter(Boolean) as CanvasEdge[]
        
        s.canvasState.nodes = [...(s.canvasState.nodes || []), ...newNodes]
        s.canvasState.edges = [...(s.canvasState.edges || []), ...newEdges]
        s.canvasState = fixEdgeReferences(s.canvasState)
        s.canvasState.history.push(s.canvasState)
      }),

        const prev = s.canvasState.history.undo()
        if (prev) s.canvasState = prev
      }),
      
      redoCanvas: () => set(s => {
        const next = s.canvasState.history.redo()
        if (next) s.canvasState = next
      }),
      
      clearCanvas: () => set(s => {
        s.canvasState = {
          nodes: [],

          viewport: { x: 0, y: 0, zoom: 1 },

          }

        } catch {

        }

          ...stroke,
          id: uid("stroke"),

        if (!id) return
        const idx = s.whiteboardStrokes.findIndex(st => st?.id === id)
        if (idx !== -1) Object.assign(s.whiteboardStrokes[idx], patch)
      }),
      
      deleteStroke: (id) => set(s => {
        if (!id) return
        s.whiteboardStrokes = s.whiteboardStrokes.filter(st => st?.id !== id)
      }),
      
      addShape: (shape) => set(s => {
        if (!shape || !shape.type) return
        s.whiteboardShapes = [...s.whiteboardShapes, {
          ...shape,
          id: uid("shape"),
          createdAt: new Date().toISOString()
        }]
      }),
      
      updateShape: (id, patch) => set(s => {
        if (!id) return
        const idx = s.whiteboardShapes.findIndex(sh => sh?.id === id)
        if (idx !== -1) Object.assign(s.whiteboardShapes[idx], patch)
      }),
      
      deleteShape: (id) => set(s => {
        if (!id) return
        s.whiteboardShapes = s.whiteboardShapes.filter(sh => sh?.id !== id)
      }),
      
      setWhiteboardTool: (tool) => set(s => { s.whiteboardTool = tool }),
      setStrokeColor: (color) => set(s => { s.strokeColor = color }),
      setStrokeWidth: (width) => set(s => { s.strokeWidth = clamp(width, 1, 20) }),
      clearWhiteboard: () => set(s => {
        s.whiteboardStrokes = []
        s.whiteboardShapes = []
      }),
