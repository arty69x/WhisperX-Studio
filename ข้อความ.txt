function R(s,a){
  switch(a.type){
    case"SET_AUTH":return{...s,authTier:a.t,authUser:a.u};
    case"LOGOUT":return{...s,authTier:0,authUser:null,module:"home"};
    case"NAV":return{...s,module:a.id};
    case"TOGGLE_PAL":return{...s,palette:!s.palette};
    case"SET_SIDEBAR":return{...s,sidebarMode:a.v};
    case"EXP_F":return{...s,expF:s.expF.includes(a.v)?s.expF.filter(x=>x!==a.v):[...s.expF,a.v]};
    case"EXP_W":return{...s,expW:s.expW.includes(a.v)?s.expW.filter(x=>x!==a.v):[...s.expW,a.v]};
    case"EXP_C":return{...s,expC:s.expC.includes(a.v)?s.expC.filter(x=>x!==a.v):[...s.expC,a.v]};
    case"SET_FILES":return{...s,files:a.v};
    case"SET_TASKS":return{...s,tasks:a.v};
    case"SET_SWOT":return{...s,swot:a.v};
    case"SET_KICKOFF":return{...s,kickoff:a.v};
    case"SET_READINESS":return{...s,readiness:a.v};
    case"SET_HISTORY":return{...s,history:a.v};
    case"CN_ADD":return{...s,canvasNodes:[...s.canvasNodes,a.n]};
    case"CN_UPD":return{...s,canvasNodes:s.canvasNodes.map(n=>n.id===a.id?{...n,...a.p}:n)};
    case"CN_DEL":return{...s,canvasNodes:s.canvasNodes.filter(n=>n.id!==a.id)};
    case"CE_ADD":return{...s,canvasEdges:[...s.canvasEdges,a.e]};
    case"CE_DEL":return{...s,canvasEdges:s.canvasEdges.filter(e=>e.id!==a.id)};
    case"CV":return{...s,canvasPan:a.pan,canvasZoom:a.zoom};
    case"LOAD_TPL":return{...s,canvasNodes:a.nodes,canvasEdges:a.edges,canvasPan:{x:0,y:0},canvasZoom:1};
    case"SET_THEME":return{...s,theme:{...s.theme,...a.v}};
    case"SET_AI":return{...s,aiProvider:a.v};
    default:return s;
  }
}

async function callAI(prompt,sys="You are WhisperX AI, a helpful project intelligence assistant."){
  const r=await fetch("https://api.anthropic.com/v1/messages",{
    method:"POST",headers:{"Content-Type":"application/json"},
    body:JSON.stringify({model:MDL,max_tokens:1000,system:sys,messages:[{role:"user",content:prompt}]}),
  });
  if(!r.ok)throw new Error(`API ${r.status}`);
  const d=await r.json();
  if(d.error)throw new Error(d.error.message);
  return d.content?.[0]?.text||"";
}

function Sparkline({data,color=G.rose}){
  if(!data||data.length<2)return null;
  const mn=Math.min(...data),mx=Math.max(...data),rng=mx-mn||1,w=80,h=26;
  const pts=data.map((v,i)=>`${(i/(data.length-1))*w},${h-((v-mn)/rng)*h}`).join(" ");
  const id=`sg${color.replace(/[^a-z0-9]/gi,"")}`;
  return<svg width={w} height={h} style={{overflow:"visible"}}>
    <defs><linearGradient id={id} x1="0" y1="0" x2="1" y2="0">
      <stop offset="0%" stopColor={color} stopOpacity="0.4"/>
      <stop offset="100%" stopColor={color}/>
    </linearGradient></defs>
    <polyline points={pts} fill="none" stroke={`url(#${id})`} strokeWidth={2} strokeLinecap="round"/>
  </svg>;
}

function CanvasEngine({state,dispatch,notify}){
  const{canvasNodes:nodes,canvasEdges:edges,canvasPan:pan,canvasZoom:zoom}=state;
  const[sel,setSel]=useState([]);const[drag,setDrag]=useState(null);const[insp,setInsp]=useState(null);

  const[showGal,setShowGal]=useState(false);const[showDL,setShowDL]=useState(false);

    const id=`n${Date.now()}`;
    dispatch({type:"CN_ADD",n:{id,x:60+Math.random()*280,y:50+Math.random()*200,w:150,h:66,label:n.label,type:n.key||n.type||"process",color:n.color||G.rose,shape:n.shape||"rect",z:nodes.length+1,load:Math.floor(Math.random()*70)}});
    notify(`Added "${n.label}" 🌸`);
  };

  const onCanvasDown=e=>{
    if(e.target!==ref.current&&!e.target.classList.contains("cb"))return;
    e.preventDefault();if(conn){setConn(null);return;}
    setSel([]);setInsp(null);
    setDrag({type:"pan",sx:e.clientX,sy:e.clientY,ox:pan.x,oy:pan.y});
  };
  const onNodeDown=(e,node)=>{
    e.stopPropagation();e.preventDefault();
    if(conn&&conn!==node.id){dispatch({type:"CE_ADD",e:{id:`e${Date.now()}`,from:conn,to:node.id}});setConn(null);notify("Connected ✓");return;}
    if(!sel.includes(node.id))setSel([node.id]);setInsp(node.id);
    dispatch({type:"CN_UPD",id:node.id,p:{z:nodes.reduce((m,n)=>Math.max(m,n.z||1),1)+1}});
    setDrag({type:"node",sx:e.clientX,sy:e.clientY,ox:node.x,oy:node.y,id:node.id});
  };
  const onResizeDown=(e,node)=>{e.stopPropagation();e.preventDefault();setDrag({type:"resize",sx:e.clientX,sy:e.clientY,ow:node.w,oh:node.h,id:node.id});};

  useEffect(()=>{
    if(!drag)return;
    const mv=e=>{
      const dx=(e.clientX-drag.sx)/zoom,dy=(e.clientY-drag.sy)/zoom;
      if(drag.type==="pan")dispatch({type:"CV",pan:{x:drag.ox+dx*zoom,y:drag.oy+dy*zoom},zoom});
      else if(drag.type==="node")dispatch({type:"CN_UPD",id:drag.id,p:{x:Math.max(0,drag.ox+dx),y:Math.max(0,drag.oy+dy)}});
      else if(drag.type==="resize")dispatch({type:"CN_UPD",id:drag.id,p:{w:Math.max(100,drag.ow+(e.clientX-drag.sx)/zoom),h:Math.max(52,drag.oh+(e.clientY-drag.sy)/zoom)}});
    };
    const up=()=>setDrag(null);
    window.addEventListener("mousemove",mv);window.addEventListener("mouseup",up);
    return()=>{window.removeEventListener("mousemove",mv);window.removeEventListener("mouseup",up);};
  },[drag,zoom,dispatch]);

  const onWheel=useCallback(e=>{
    e.preventDefault();
    const nz=Math.max(0.12,Math.min(4,zoom*(e.deltaY>0?0.92:1.08)));
    dispatch({type:"CV",pan,zoom:+nz.toFixed(3)});
  },[zoom,pan,dispatch]);

  const inspNode=insp?nodes.find(n=>n.id===insp):null;
  const avgLoad=nodes.length?Math.round(nodes.reduce((a,n)=>a+(n.load||0),0)/nodes.length):0;

  return(
    <div style={{display:"flex",height:"100%",gap:0}}>
      <div style={{flex:1,display:"flex",flexDirection:"column",gap:8,overflow:"hidden"}}>
        {/* Toolbar */}
        <GC style={{padding:"7px 12px",borderRadius:14,display:"flex",alignItems:"center",gap:7,flexWrap:"wrap",flexShrink:0}}>
          <span style={{fontSize:11,fontFamily:G.fD,fontWeight:700,color:G.pink,marginRight:2}}>Canvas Engine</span>
          <div style={{display:"flex",gap:3}}>
            {[{I:ZoomIn,fn:()=>dispatch({type:"CV",pan,zoom:Math.min(4,+(zoom+0.15).toFixed(2))})},
              {I:ZoomOut,fn:()=>dispatch({type:"CV",pan,zoom:Math.max(0.12,+(zoom-0.15).toFixed(2))})},
            ].map(({I,fn},i)=><GB key={i} onClick={fn} sm style={{padding:"4px 7px",borderRadius:8}}><I size={11}/></GB>)}
          </div>
          <span style={{fontSize:8,fontFamily:G.fM,color:G.muted,padding:"2px 7px",background:G.glass,borderRadius:8,border:`1px solid ${G.border}`}}>{Math.round(zoom*100)}%</span>
